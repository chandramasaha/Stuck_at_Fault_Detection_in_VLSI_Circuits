import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

# Load the results of each autoencoder
stacked_results = pd.read_csv("C:/Users/User/FaultDetection_SAE.csv")
sparse_results = pd.read_csv("C:/Users/User/FaultDetection_SpAE.csv")
vae_results = pd.read_csv("C:/Users/User/FaultDetection_VAE.csv")
ssae_results = pd.read_csv("C:/Users/User/FaultDetection_SSAE.csv")

# Extract the precomputed reconstruction error column
stacked_errors = stacked_results["Reconstruction Error"]
sparse_errors = sparse_results["Reconstruction Error"]
vae_errors = vae_results["Reconstruction Error"]
ssae_errors = ssae_results["Reconstruction Error"]

# Plot histograms of reconstruction errors
plt.figure(figsize=(10, 5))
plt.hist(stacked_errors, bins=50, alpha=0.6, label="Stacked Autoencoder", color="black")
plt.hist(sparse_errors, bins=50, alpha=0.6, label="Sparse Autoencoder", color="blue")
plt.hist(vae_errors, bins=50, alpha=0.6, label="Variational Autoencoder", color="red")
plt.hist(ssae_errors, bins=50, alpha=0.6, label="Stacked Sparse Autoencoder (SSAE)", color="green")
plt.xlabel("Reconstruction Error")
plt.ylabel("Frequency")
plt.legend()
plt.title("Comparison of Reconstruction Errors")
plt.show()

# Ensure reconstruction errors are a NumPy array and flatten if necessary
stacked_errors = np.array(stacked_errors).flatten()  # Convert & flatten

# Check the shape (debugging)
print("Shape of stacked_errors:", stacked_errors.shape)  

# Set a threshold (mean + std as an example)
threshold = np.mean(stacked_errors) + 2*np.std(stacked_errors)

# Assign colors based on threshold
colors = ['green' if err < threshold else 'red' for err in stacked_errors]

# Scatter plot
plt.figure(figsize=(8,5))
plt.scatter(range(len(stacked_errors)), stacked_errors, c=colors, alpha=0.7)

# Plot threshold line
plt.axhline(threshold, color='blue', linestyle='dashed', linewidth=2, label="Threshold")

plt.xlabel("Sample Index")
plt.ylabel("Reconstruction Error")
plt.title("Stacked Autoencoder Scatter Plot")
plt.legend(["Threshold", "Normal (Green)", "Faulty (Red)"])
plt.show()
