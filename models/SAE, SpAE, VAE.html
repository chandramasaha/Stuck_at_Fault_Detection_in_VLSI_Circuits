#Stacked Autoencoder (SAE)
    
import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow.keras.models import Model
from tensorflow.keras.layers import Input, Dense, Dropout, BatchNormalization
from tensorflow.keras.callbacks import EarlyStopping
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
import matplotlib.pyplot as plt

# Load the dataset
df = pd.read_csv('C:/Users/User/OneDrive/Desktop/u n i/SEM #7/Major Project/work/updated_dataset.csv')

# Convert binary strings to numerical format
for col in df.columns:
    if df[col].dtype == 'object':  
        df[col] = df[col].apply(lambda x: [int(bit) for bit in x.strip() if bit in '01']) 
        df[col] = df[col].apply(lambda x: sum(b * (2**i) for i, b in enumerate(reversed(x))))  

# Normalize the data
scaler = StandardScaler()
data_scaled = scaler.fit_transform(df.drop(columns=["Circuit-Name"]))

# Split into training and testing sets
X_train, X_test = train_test_split(data_scaled, test_size=0.25, random_state=42)

# Define Stacked Autoencoder
input_layer = Input(shape=(X_train.shape[1],))
encoded = Dense(128, activation='relu')(input_layer)
encoded = Dense(64, activation='relu')(encoded)
encoded = Dense(32, activation='relu')(encoded)  # Bottleneck layer
decoded = Dense(64, activation='relu')(encoded)
decoded = Dense(128, activation='relu')(decoded)
decoded = Dense(X_train.shape[1], activation='linear')(decoded)

stacked_autoencoder = Model(inputs=input_layer, outputs=decoded)
stacked_autoencoder.compile(optimizer='adam', loss='mae')
stacked_autoencoder.summary()

# Train the autoencoder
history = stacked_autoencoder.fit(
    X_train, X_train,
    epochs=100,
    batch_size=64,
    shuffle=True,
    validation_data=(X_test, X_test),
    callbacks=[EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)]
)

# Fault detection using reconstruction errors
reconstructed = stacked_autoencoder.predict(data_scaled)
errors = np.mean(np.abs(data_scaled - reconstructed), axis=1)
threshold = np.percentile(errors, 75)  # Threshold for anomaly detection

df["Fault Detected"] = errors > threshold
df.to_csv("SAE_results.csv", index=False)
print("Stacked Autoencoder results saved!")

#Variational Autoencoder
import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow.keras.models import Model
from tensorflow.keras.layers import Input, Dense, Lambda, Layer
from tensorflow.keras import backend as K
from tensorflow.keras.callbacks import EarlyStopping
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split

# Load and preprocess dataset
df = pd.read_csv("C:/Users/User/OneDrive/Desktop/u n i/SEM #7/Major Project/work/updated_dataset.csv")

# Convert binary strings to numerical format
for col in df.columns:
    if df[col].dtype == 'object':  
        df[col] = df[col].apply(lambda x: [int(bit) for bit in x.strip() if bit in '01']) 
        df[col] = df[col].apply(lambda x: sum(b * (2**i) for i, b in enumerate(reversed(x))))  

# Normalize the data
scaler = StandardScaler()
data_scaled = scaler.fit_transform(df.drop(columns=["Circuit-Name"]))

# Split into training and testing sets
X_train, X_test = train_test_split(data_scaled, test_size=0.25, random_state=42)

# Define Sampling Function for VAE
def sampling(args):
    z_mean, z_log_var = args
    batch = K.shape(z_mean)[0]
    dim = K.int_shape(z_mean)[1]
    epsilon = K.random_normal(shape=(batch, dim))
    return z_mean + K.exp(0.5 * z_log_var) * epsilon

# Define Variational Autoencoder
latent_dim = 32

# Encoder
inputs = Input(shape=(X_train.shape[1],))
x = Dense(128, activation='relu')(inputs)
x = Dense(64, activation='relu')(x)
z_mean = Dense(latent_dim, name='z_mean')(x)
z_log_var = Dense(latent_dim, name='z_log_var')(x)
z = Lambda(sampling, output_shape=(latent_dim,), name='z')([z_mean, z_log_var])

encoder = Model(inputs, [z_mean, z_log_var, z], name='encoder')

# Decoder
decoder_input = Input(shape=(latent_dim,))
x = Dense(64, activation='relu')(decoder_input)
x = Dense(128, activation='relu')(x)
outputs = Dense(X_train.shape[1], activation='linear')(x)
decoder = Model(decoder_input, outputs, name='decoder')

# Custom VAE loss layer
class VAELossLayer(Layer):
    def call(self, inputs):
        y_true, y_pred, z_mean, z_log_var = inputs
        reconstruction_loss = tf.keras.losses.mae(y_true, y_pred)
        kl_loss = -0.5 * K.sum(1 + z_log_var - K.square(z_mean) - K.exp(z_log_var), axis=-1)
        total_loss = K.mean(reconstruction_loss + kl_loss)
        self.add_loss(total_loss)  # Correct way to add loss inside a layer
        return y_pred  # Return y_pred so that the model output remains the same

# Define VAE model
z_mean_out, z_log_var_out, z_out = encoder(inputs)
vae_outputs = decoder(z_out)

# Apply the custom loss layer
vae_outputs = VAELossLayer()([inputs, vae_outputs, z_mean_out, z_log_var_out])

# Final VAE model
vae = Model(inputs, vae_outputs, name='vae')

# Compile VAE
vae.compile(optimizer='adam')

vae.summary()

# Train the VAE
vae.fit(
    X_train, X_train,
    epochs=100,
    batch_size=64,
    shuffle=True,
    validation_data=(X_test, X_test),
    callbacks=[EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)]
)

# Fault detection using reconstruction error
reconstructed = vae.predict(data_scaled)
errors = np.mean(np.abs(data_scaled - reconstructed), axis=1)
threshold = np.percentile(errors, 75)

df["Fault Detected"] = errors > threshold
df.to_csv("VAE_results.csv", index=False)
print("Variational Autoencoder results saved!")
