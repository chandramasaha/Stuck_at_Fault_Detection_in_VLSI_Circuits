# Convert binary strings to numerical arrays
for col in df.columns:
    if df[col].dtype == 'object':
        df[col] = df[col].apply(lambda x: [int(bit) for bit in x.strip() if bit in '01'])
        df[col] = df[col].apply(lambda x: sum(b * (2**i) for i, b in enumerate(reversed(x))))

# Define the Stacked Sparse Autoencoder (SSAE)
input_layer = Input(shape=(X_train.shape[1],))
encoded = Dense(128, activation='relu', activity_regularizer=l1(1e-5))(input_layer)
encoded = Dropout(0.2)(encoded)
encoded = BatchNormalization()(encoded)
encoded = Dense(64, activation='relu', activity_regularizer=l1(1e-5))(encoded)
bottleneck = Dense(32, activation='relu', activity_regularizer=l1(1e-5))(encoded)
decoded = Dense(64, activation='relu')(bottleneck)
decoded = Dropout(0.2)(decoded)
decoded = Dense(128, activation='relu')(decoded)
decoded = Dense(X_train.shape[1], activation='linear')(decoded)

# Compile the SSAE model
ssae = Model(inputs=input_layer, outputs=decoded)
ssae.compile(optimizer='adam', loss='mae')
ssae.summary()

# Train the SSAE
early_stopping = EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)
history = ssae.fit(
    X_train, X_train,
    epochs=100,
    batch_size=64,
    shuffle=True,
    validation_data=(X_test, X_test),
    callbacks=[early_stopping]
)

# Compute reconstruction errors
reconstructed = ssae.predict(data_scaled)
reconstruction_errors = np.mean(np.abs(data_scaled - reconstructed), axis=1)

# Set fault detection threshold
threshold = np.percentile(reconstruction_errors, 75)
fault_labels = reconstruction_errors > threshold

# Save results
df_results = pd.DataFrame({
    "Circuit Name": circuit_names,
    "Reconstruction Error": reconstruction_errors,
    "Fault Detected": fault_labels
})
df_results.to_csv("circuit_fault_detection_results_ssae.csv", index=False)
